Create **OsvStatusBarWidgetFactory.kt** in `src/main/java/io/dyuti/osvplugin/statusbar/`:

```kotlin
package io.dyuti.osvplugin.statusbar

import com.intellij.openapi.project.Project
import com.intellij.openapi.wm.StatusBar
import com.intellij.openapi.wm.StatusBarWidget
import com.intellij.openapi.wm.StatusBarWidgetFactory
import com.intellij.util.Consumer
import java.awt.Component
import java.awt.event.MouseEvent
import javax.swing.JLabel

class OsvStatusBarWidgetFactory : StatusBarWidgetFactory {
    override fun getId(): String = "osv.statusBar"
    override fun getDisplayName(): String = "OSV Scanner"
    override fun isAvailable(project: Project): Boolean = true
    override fun createWidget(project: Project): StatusBarWidget = OsvStatusBarWidget(project)
    override fun disposeWidget(widget: StatusBarWidget) {}
    override fun canBeEnabledOn(statusBar: StatusBar): Boolean = true
}

class OsvStatusBarWidget(private val project: Project) : StatusBarWidget {
    @Volatile private var text: String = "OSV: Ready"
    private val label = JLabel(text)

    override fun ID(): String = "osv.statusBar.widget"
    override fun getPresentation(): StatusBarWidget.WidgetPresentation = object : StatusBarWidget.TextPresentation {
        override fun getText(): String = text
        override fun getTooltipText(): String = "OSV Vulnerability Scanner status"
        override fun getClickConsumer(): Consumer<MouseEvent>? = null
        override fun getAlignment(): Float = Component.LEFT_ALIGNMENT
    }
    override fun install(statusBar: StatusBar) {}
    override fun dispose() {}

    fun updateStatus(newText: String) {
        text = newText
        label.text = newText
    }
}
```

In plugin.xml, add inside `<extensions defaultExtensionNs="com.intellij">`:

```xml
<statusBarWidgetFactory implementation="io.dyuti.osvplugin.statusbar.OsvStatusBarWidgetFactory" id="osv.statusBar"/>
```

In OsVToolWindowPanel.kt, modify `updateStatus()` (or create it if it doesn't exist) to also update the status bar widget:

```kotlin
}
```
